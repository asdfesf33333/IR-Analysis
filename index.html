Absenteeism Dashboard

Automated Factor Analysis & Weekly Patterns • 2026

Click to Upload Excel/CSV

Weekday Absenteesim Pattern

Factor Analysis loading Table

""" # --- BACKEND LOGIC --- app = Flask(__name__) MONTH_MAP = {m.lower(): i for i, m in enumerate(calendar.month_name) if m} def get_weekday(day, month_name): try: month_num = MONTH_MAP.get(month_name.lower()) d = date(YEAR, month_num, int(day)) return d.strftime('%A') except: return None @app.route('/') def index(): return render_template_string(INDEX_HTML) @app.route('/upload', methods=['POST']) def upload_file(): file = request.files['file'] filepath = os.path.join(UPLOAD_FOLDER, file.filename) file.save(filepath) try: data_dict = pd.read_excel(filepath, sheet_name=None) scatter_list = [] matrix_data = [] for sheet, df in data_dict.items(): df.columns = [str(c).strip() for c in df.columns] id_col = next((c for c in df.columns if 'USERID' in c.upper()), 'USERID') name_col = next((c for c in df.columns if 'NAME' in c.upper()), 'NAME') date_cols = [c for c in df.columns if c.isdigit() and 1 <= int(c) <= 31] for _, row in df.iterrows(): absences = [] for d in date_cols: is_present = (str(row[d]).strip().upper() == 'P-P') weekday = get_weekday(d, sheet) if not is_present: scatter_list.append({'Weekday': weekday}) absences.append({'key': f"{sheet}_{d}", 'val': 0 if is_present else 1}) matrix_data.append(absences) # Factor Analysis all_keys = sorted(list(set(d['key'] for entry in matrix_data for d in entry))) rows = [[{d['key']: d['val'] for d in e}.get(k, 0) for k in all_keys] for e in matrix_data] fa_df = pd.DataFrame(rows, columns=all_keys) fa_df = fa_df.loc[:, (fa_df != fa_df.iloc[0]).any()] # filter no variance factor_table = [] if not fa_df.empty and fa_df.shape[0] > fa_df.shape[1]: fa = FactorAnalyzer(rotation="varimax") fa.fit(fa_df) ev, _ = fa.get_eigenvalues() n_factors = sum(ev > 1) if n_factors > 0: fa = FactorAnalyzer(n_factors=n_factors, rotation="varimax") fa.fit(fa_df) loading_df = pd.DataFrame(fa.loadings_, columns=[f"F{i+1}" for i in range(n_factors)], index=fa_df.columns) factor_table = loading_df.reset_index().round(3).to_dict(orient='records') return jsonify({'scatter_data': scatter_list, 'factor_table': factor_table}) except Exception as e: return jsonify({'error': str(e)}), 500 if __name__ == '__main__': print(f"Launching Dashboard on http://127.0.0.1:{PORT}") app.run(debug=True, port=PORT)
